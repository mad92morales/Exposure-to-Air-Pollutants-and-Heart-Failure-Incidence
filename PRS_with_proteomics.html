# Extract model summary and confidence intervals
coef_all <- summary(model_all_prs_proteins)$coefficients
conf_all <- confint(model_all_prs_proteins)

# Build data frame for plotting
df_all_forest <- data.frame(
  Variable = rownames(coef_all),
  OR = exp(coef_all[, "Estimate"]),
  LowerCI = exp(conf_all[, 1]),
  UpperCI = exp(conf_all[, 2]),
  p_value = coef_all[, "Pr(>|z|)"]
)

# Filter out intercept
df_all_forest <- df_all_forest[df_all_forest$Variable != "(Intercept)", ]

# Add significance indicator
df_all_forest$Significant <- ifelse(df_all_forest$p_value < 0.05, "Yes", "No")

# Plot using ggplot2
library(ggplot2)

m <- ggplot(df_all_forest, aes(x = reorder(Variable, OR), y = OR, color = Significant)) +
  geom_point(size = 3) +
  geom_errorbar(aes(ymin = LowerCI, ymax = UpperCI), width = 0.2) +
  geom_hline(yintercept = 1, linetype = "dashed", color = "gray50") +
  coord_flip() +
  scale_y_log10() +
  scale_color_manual(values = c("Yes" = "steelblue", "No" = "gray")) +
  labs(title = "Forest Plot  PRS + Proteins Model",
       y = "Odds Ratio (log scale)", x = "") +
  theme_minimal()

ggsave(
  filename = "Figures/forest_plot_all_prs.png",
  plot = m,       
  width = 3,                     
  height = 3,
  dpi = 300)
