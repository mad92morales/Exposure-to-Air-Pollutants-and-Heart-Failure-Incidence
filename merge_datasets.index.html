# Merge Datasets

# 1. All 3 datasets: prot (original), pheno_clean, env_clean
merged_all <- merge(pheno_clean, env_clean, by = "participant_id", all = FALSE)
merged_all <- merge(merged_all, prot, by = "participant_id", all = FALSE)

# 2. All 3 datasets (complete-case): prot_no_na, pheno_no_na, env_no_na
merged_all_no_na <- merge(pheno_no_na, env_no_na, by = "participant_id", all = FALSE)
merged_all_no_na <- merge(merged_all_no_na, prot_no_na, by = "participant_id", all = FALSE)

# 3. Phenotype + pollution (cleaned, not filtered for NA)
merged_pheno_env <- merge(pheno_clean, env_clean, by = "participant_id", all = FALSE)

# 4. Phenotype + pollution (no NA)
merged_pheno_env_no_na <- merge(pheno_no_na, env_no_na, by = "participant_id", all = FALSE)
